//filter:不会操作原数组,返回过滤后的新数组,回调函数的返回结果:如果返回true,表示这一项放到新数组中
let newArray = [1,2,3,4,5].filter(function(index) {
    return index>2 && index<5;
});
console.log(newArray);//[3,4]
