//将原数组中每个元素加 4 转换为一个新数组
var arr1 = $.map( [0,1,2], function(n){
  return n + 4;
});
console.log(arr1);//[4,5,6]


//原数组中大于 0 的元素加 1 ，否则删除
var arr2 = $.map( [0,1,2], function(n){
  return n > 0 ? n + 1 : null;
});
console.log(arr2);//[2,3]


//原数组中每个元素扩展为一个包含其本身和其值加 1 的数组，并转换为一个新数组
var arr3 = $.map( [0,1,2], function(n){
  return [ n, n + 1 ];
});
console.log(arr3);//[0, 1, 1, 2, 2, 3]
