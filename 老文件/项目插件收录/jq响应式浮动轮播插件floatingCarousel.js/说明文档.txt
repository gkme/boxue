$('#carousel-default').floatingCarousel();

自适应轮播创建：
var opts = {
    // options here
},
responsiveCarousel = $('#carousel-default').floatingCarousel(opts);
$(window).resize($.debounce(100, function () {
    responsiveCarousel.update(opts);
}));

所有配置选项：
$('#carousel-default').floatingCarousel({
      autoScroll : false,  // auto scroll
      autoScrollDirection : 'left', // or 'right'
      autoScrollSpeed : 10000,// scroll speed in milliseconds
      initClass: 'floatingCarouselContainer',// initial CSS class
      looped : true,// infinite loop
      scrollerAlignment : 'horizontal',// or 'vertical'
      scrollerOffset : 0,// scroll offset in pixels
      scrollSpeed : 'medium',// or 'slow', 'fast'
      enableTouchEvents : true,// enable touch events
      touchOverflowHidden : true,// prevent unwanted page scrolling during touch events
      reverseOnTouch : true  // reverse the the direction on touch
});


回调函数：
$('#carousel-default').floatingCarousel({
    beforeCreateFunction : null,
    afterCreateFunction : null
})
